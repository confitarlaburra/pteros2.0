124 inline spdlog::level::level_enum from_str(const std::string &name) SPDLOG_NOEXCEPT
125 {
126  static std::unordered_map<std::string, level_enum> name_to_level = // map string->level
127  {{level_names[0], level::trace}, // trace
128  {level_names[1], level::debug}, // debug
129  {level_names[2], level::info}, // info
130  {level_names[3], level::warn}, // warn
131  {level_names[4], level::err}, // err
132  {level_names[5], level::critical}, // critical
133  {level_names[6], level::off}}; // off
134 
135  auto lvl_it = name_to_level.find(name);
136  return lvl_it != name_to_level.end() ? lvl_it->second : level::off;
137 }
