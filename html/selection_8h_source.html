773  template<class F>
774  void split(std::vector<Selection>& parts, F callback){
775  using namespace std;
776  parts.clear();
777  using Ret = decltype(callback(*this,0));
778  // Map
779  map<Ret,vector<int> > m;
780  for(int i=0; i<size(); ++i){
781  m[callback(*this,i)].push_back(index(i));
782  }
783  // Create selections
784  typename map<Ret,vector<int> >::iterator it;
785  for(it=m.begin();it!=m.end();it++){
786  parts.emplace_back(*system, it->second.begin(), it->second.end());
787  }
788  }
